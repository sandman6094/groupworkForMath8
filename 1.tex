    \textbf{Error Bounds:} \\\\
    \begin{align*}
       |f^{"}(x)| &= |(f^{'}(x))^{'}| \\
        &= |(2x \, cos(x^2))^{'}| \\
        &= |-4x^2\,sin(x^2)+2\,cos(x^2)| \\
        &= 4x^2|sin(x^2)| + 2|cos(x^2)|
    \end{align*}

    Because $-1 \leq x \leq 1$, then:
    \begin{align*}
        |f^{"}(x)| & \leq 2
    \end{align*}
    We have $K = 2$, then for Midpoint Rule:
    \begin{align*}
        |E_{M}| &\leq \frac{K(b-a)^{3}}{24n^2} \\
        &\leq \frac{2\,(2)^3}{24n^2} \\
        &\leq \frac{2}{3n^2}
    \end{align*} \\
